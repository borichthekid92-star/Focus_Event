            });

            // Highlight fields in details
            $("#events tbody td.details").each(function() {
                let preview = $(this).find('.details-preview').html();
                let full = $(this).find('.details-full').html();
                
                // Apply highlighting
                preview = preview.replace(/([A-Za-z_]+):/g, "<strong>$1:</strong>");
                full = full.replace(/([A-Za-z_]+):/g, "<strong>$1:</strong>");
                
                $(this).find('.details-preview').html(preview);
                $(this).find('.details-full').html(full);
            });

            // Modal functionality
            const modal = document.getElementById('eventModal');
            const closeBtn = document.getElementsByClassName('close-btn')[0];

            // Click event for table rows
            $('#events tbody').on('click', 'tr', function() {
                const rowData = table.row(this).data();
                showEventDetails(rowData);
            });

            function showEventDetails(rowData) {
                const timestamp = rowData[0];
                const eventType = rowData[1];
                const details = $(rowData[2]).find('.details-full').text() || $(rowData[2]).text();
                
                // Parse the details string
                const eventDetails = parseEventDetails(details);
                
                // Build the detail grid
                let gridHtml = '';
                gridHtml += `<div class="detail-label">Timestamp</div>`;
                gridHtml += `<div class="detail-value">${timestamp}</div>`;
                gridHtml += `<div class="detail-label">Event Type</div>`;
                gridHtml += `<div class="detail-value">${eventType}</div>`;
                
                // Add parsed details
                for (const [key, value] of Object.entries(eventDetails)) {
                    gridHtml += `<div class="detail-label">${formatLabel(key)}</div>`;
                    gridHtml += `<div class="detail-value">${value}</div>`;
                }
                
                document.getElementById('eventDetailGrid').innerHTML = gridHtml;
                document.getElementById('rawDetailsContent').textContent = details;
                
                modal.style.display = 'block';
            }

            function parseEventDetails(detailsStr) {
                const details = {};
                // Split by | and then by :
                const parts = detailsStr.split('|');
                
                parts.forEach(part => {
                    // Look for key:value patterns
                    const match = part.match(/([A-Za-z_]+):\s*(.+?)(?=\s*[A-Za-z_]+:|$)/);
                    if (match) {
                        const key = match[1].trim();
                        const value = match[2].trim();
                        details[key] = value;
                    } else {
                        // Handle parts without clear key:value structure
                        const colonIndex = part.indexOf(':');
                        if (colonIndex > -1) {
                            const key = part.substring(0, colonIndex).trim();
                            const value = part.substring(colonIndex + 1).trim();
                            if (key && value) {
                                details[key] = value;
                            }
                        }
                    }
                });
                
                return details;
            }

            function formatLabel(key) {
                // Convert snake_case or camelCase to Title Case
                return key.replace(/_/g, ' ')
                    .replace(/([A-Z])/g, ' $1')
                    .trim()
                    .replace(/\b\w/g, l => l.toUpperCase());
            }

            // Close modal

                    const originalHtml = copyBtn.innerHTML;

                    setTimeout(() => {

                    }, 2000);
                });
            }

            // Time filter buttons

                    const rowDate = new Date(timestamp);
                    const diffHrs = (now - rowDate) / 36e5;
                    $(this.node()).toggle(diffHrs <= hours);
                });
            });

            // --- Process Sequence Detection ---
            let rule = null;
            let recentProcs = [];

            $("#saveRule").click(function() {
                const p1 = $("#proc1").val().trim().toLowerCase();
                const p2 = $("#proc2").val().trim().toLowerCase();
                const p3 = $("#proc3").val().trim().toLowerCase();
                const windowMin = parseInt($("#timeWindow").val());
                if (!p1 || !p2 || !p3) {
                    alert("Enter all 3 process names!");
                    return;
                }
                rule = { p1, p2, p3, windowMin };

            });

            // Scan for sequence function
            function scanForSequence() {
                if (!rule) return;
                const rows = table.rows({ filter: 'applied' }).data();
                const now = new Date();

                recentProcs = [];
                for (let i = 0; i < rows.length; i++) {
                    const details = $(rows[i][2]).text().toLowerCase();
                    const timestamp = new Date(rows[i][0]);
                    const ageMin = (now - timestamp) / 60000;
                    if (ageMin <= rule.windowMin) {
                        if (details.includes(rule.p1)) recentProcs.push(rule.p1);
                        if (details.includes(rule.p2)) recentProcs.push(rule.p2);
                        if (details.includes(rule.p3)) recentProcs.push(rule.p3);
                    }
                }

                if (rule && recentProcs.includes(rule.p1) &&
                    recentProcs.includes(rule.p2) &&
                    recentProcs.includes(rule.p3)) {
                    showTempAlert(`⚠️ Sequence detected: ${rule.p1}, ${rule.p2}, ${rule.p3} executed within ${rule.windowMin} min`);
                    recentProcs = [];
                }
            }

            // --- Single Process Risk Assessment ---
            $("#calculateRisk").click(function() {
                const riskProc = $("#riskProc").val().trim().toLowerCase();
                if (!riskProc) {
                    alert("Enter a process name!");
                    return;
                }
                const rows = table.rows({ filter: 'applied' }).data();
                let procCount = 0;
                let totalEvents = rows.length;

                for (let i = 0; i < totalEvents; i++) {
                    const details = $(rows[i][2]).text().toLowerCase();
                    if (details.includes(riskProc)) {
                        procCount++;
                    }
                }

                const riskRatio = totalEvents > 0 ? (procCount / totalEvents) : 0;
                const riskPercent = riskRatio * 100;

                if (riskPercent < 5) {
                    showTempAlert(`⚠️ Low Risk Alert: ${riskProc} appears in only ${riskPercent.toFixed(2)}% of events (${procCount}/${totalEvents})`, 30000);
                } else {
                    showTempAlert(`Risk Assessment: ${riskProc} appears in ${riskPercent.toFixed(2)}% of events (${procCount}/${totalEvents}) - No alert`);
                }
            });

            // --- Update Risk Stats Tables ---
            function updateRiskStats() {
                $.get("/api/risk-stats", function(data) {
                    // Update Rarely seen (<1%) - Max 5
                    let lowHtml = "";
                    data.low_risk.slice(0, 5).forEach(item => {
                        lowHtml += `<tr><td>${item.process}</td><td>${item.count}</td><td>${item.ratio}%</td></tr>`;
                    });
                    $("#lowRiskTable tbody").html(lowHtml);

                    // Update Processes < 10% - Max 5
                    let mediumHtml = "";
                    data.medium_risk.slice(0, 5).forEach(item => {
                        mediumHtml += `<tr><td>${item.process}</td><td>${item.count}</td><td>${item.ratio}%</td></tr>`;
                    });
                    $("#mediumRiskTable tbody").html(mediumHtml);

                    // Average (<30%) - Max 5
                    let highHtml = "";
                    data.high_risk.slice(0, 5).forEach(item => {
                        highHtml += `<tr><td>${item.process}</td><td>${item.count}</td><td>${item.ratio}%</td></tr>`;
                    });
                    $("#highRiskTable tbody").html(highHtml);
                }).fail(function() {
                    console.error("Failed to fetch risk stats");
                });
            }

            function showTempAlert(msg, duration = 60000) {
                const alertBox = $("#alertBanner");
                alertBox.text(msg).fadeIn();
                setTimeout(() => alertBox.fadeOut(), duration);
            }

            // Scan every 10 seconds for sequence
            setInterval(scanForSequence, 10000);

            updateRiskStats();
            setInterval(updateRiskStats, 10000);
        });
